    <script>
        const hamburger = document.querySelector('.hamburger');
        const navLinks = document.querySelector('nav ul');
        
        hamburger.addEventListener('click', () => {
            navLinks.classList.toggle('active');
        });

        window.addEventListener('scroll', () => {
            const nav = document.querySelector('nav');
            const navLinks = nav.querySelectorAll('ul li a');
            const logo = nav.querySelector('.logo');
            const intro = document.querySelector('header.intro');
            const h1Elements = intro.querySelectorAll('h1');
            const subtext = intro.querySelector('.subtext');
            const arrowDown = intro.querySelector('.arrow-down');

            const scrollPos = window.scrollY;
            const introHeight = intro.offsetHeight;
            const scrollPercentage = Math.min(scrollPos / introHeight, 1);

            const bgColorValue = Math.round(255 - (255 * scrollPercentage));
            const textColorValue = Math.round(255 * scrollPercentage);
            const backgroundColor = `rgb(${bgColorValue}, ${bgColorValue}, ${bgColorValue})`;
            const textColor = `rgb(${textColorValue}, ${textColorValue}, ${textColorValue})`;

            intro.style.backgroundColor = backgroundColor;
            h1Elements.forEach(h1 => h1.style.color = textColor);
            subtext.style.color = textColor;
            arrowDown.style.color = textColor;
            nav.style.backgroundColor = scrollPercentage > 0 ? '#444' : 'white';
            logo.style.color = scrollPercentage > 0 ? 'lightgray' : 'black';
            navLinks.forEach(link => link.style.color = scrollPercentage > 0 ? 'lightgray' : 'gray');

            const sections = document.querySelectorAll('main section');

            sections.forEach((section, index) => {
                if (section.offsetTop <= scrollPos && section.offsetTop + section.offsetHeight > scrollPos) {
                    navLinks.forEach(link => link.classList.remove('active'));
                    navLinks[index].classList.add('active');
                }
            });
        });
    </script>
